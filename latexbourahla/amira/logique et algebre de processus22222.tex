\begin{enumerate}
\item  A[]B = B[]A $\to$ A \:\:\:Commutativité 
\begin{enumerate}
    \item[$\ast$]  on a A[]B \: \stackrel{a}{\rightarrow}\:  A \: \: et \: \:  B[]A \:  \stackrel{a}{\rightarrow}  \: A \:alors\: A[]B = B[]A
\end{enumerate} 
\item A[]stop = stop[]A = A $\to$\:\:\: A Zéro absorption
\begin{enumerate}
    \item[$\ast$]  on a A[]stop \: \stackrel{a}{\rightarrow}\:  A \: \: et \: \:  stop[]A \:  \stackrel{a}{\rightarrow}  \: A \:alors\: stop[]A = A[]stop=A
\end{enumerate} 
\item  A[](B[]C) = (A[]B)[]C $\to$ \:\:\:Associativité 
\begin{enumerate}
  on \: a \:  B[]C \:  \stackrel{a}{\rightarrow}  \: B \:        (1)\\
  on \: a \:  A[]B \:  \stackrel{a}{\rightarrow}  \: B \:          (2)\\   
    depuis\: (1)\: et \:(2) \:on \:trouve\: que:\: A[](B[]C) \: \stackrel{a}{\rightarrow}\: B \: \:     \:\:\:\:    (A)    \\
    et (A[]B)[]C \: \stackrel{a}{\rightarrow}\: B \: \:    \:\:\:       (B) \\
   depuis\: (A) \:et \:(B) \:on \:trouve \:que:\: A[](B[]C)=(A[]B)[]C
\end{enumerate} 
    \end{enumerate}

\subsection*{ la spécification lotos pour le circuit logique OR}
spec OR [a,b,c] :noexit \\
type BIT is \\
    sorts BIT\\
    opns 0 (*! constructor *), \\
        1 (*! constructor *) : \rightarrow BIT\\
    or : BIT ,BIT\rightarrow BIT\\
    eqns \\
        ofsort BIT \\
        or (0,0) = 0; \\
        or (0,1) = 1; \\
        or (0,1) = 1; \\
        or (1,1) = 1; \\
endtype\\ 
behaviour\\ 
    gate OR[a, b, c]\\
where \\
    process\: gate-OR[a, b] : noexit :=\\ 
    a\:?aa:Bit;\: b\:?bb:Bit; \: c\:!or(aa,bb);\: stop \\
    endproc \\
endspec\\
